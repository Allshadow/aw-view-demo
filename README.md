# aw-view

### use

```
npm install aw-view
```

#### 全局引入

```

import awView from 'aw-view'

Vue.use(awView)


```

```
<aw-demo></aw-demo>
```

#### 按需引入


```

import { awDemo } from 'aw-view'

Vue.use(awDemo)


```

```
<aw-demo></aw-demo>
```
